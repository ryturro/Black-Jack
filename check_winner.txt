def check_winner(player, dealer):
    if player.score > 21:
        return "Player busts! Dealer wins."
    elif dealer.score > 21:
        return "Dealer busts! Player wins."
    elif player.score == dealer.score:
        return "It's a tie!"
    elif player.score > dealer.score:
        return "Player wins!"
    else:
        return "Dealer wins!"
